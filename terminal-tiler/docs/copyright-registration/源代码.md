print_usage() {
  cat <<'EOF'
Usage:
  terminal-tile-all
  terminal-tile-all ZONE

Zones:
  zl2  zl3  zl4
  zr2  zr3  zr4
EOF

zone_layout_from_arg() {
  local key="$1"
  key="${key//[[:space:]]/}"
  key="${key//-/}"
  key="${key//_/}"
  key="${(L)key}"

  case "$key" in
    zl2)
      echo "term-left-half"
      return 0
      ;;
    zl3)
      echo "term-left-third"
      return 0
      ;;
    zl4)
      echo "term-left-quarter"
      return 0
      ;;
    zr2)
      echo "term-right-half"
      return 0
      ;;
    zr3)
      echo "term-right-third"
      return 0
      ;;
    zr4)
      echo "term-right-quarter"
      return 0
      ;;
    *)
      return 1
      ;;
  esac
}

looks_like_zone_arg() {
  local key="$1"
  key="${key//[[:space:]]/}"
  key="${key//-/}"
  key="${key//_/}"
  key="${(L)key}"
  case "$key" in
    zl2|zl3|zl4|zr2|zr3|zr4)
      return 0
      ;;
    *)
      return 1
      ;;
  esac
}

bootstrap_hotkey_once() {
  if [[ "${TILE_SKIP_HOTKEY_BOOTSTRAP:-0}" == "1" ]]; then
    return
  fi

  local conf_dir="$HOME/.config/terminal-window-tiler"
  local sentinel="${conf_dir}/hotkey-bootstrap-v1"
  if [[ -f "$sentinel" ]]; then
    return
  fi

  local hotkey_cmd=""
  hotkey_cmd="$(command -v terminal-tile-hotkey 2>/dev/null || true)"
  if [[ -z "$hotkey_cmd" ]]; then
    local script_dir="${0:A:h}"
    if [[ ! -x "$script_dir/terminal-tile-hotkey" ]]; then
      local x_file="${(%):-%x}"
      script_dir="${x_file:A:h}"
    fi
    if [[ -x "$script_dir/terminal-tile-hotkey" ]]; then
      hotkey_cmd="$script_dir/terminal-tile-hotkey"
    fi
  fi

  if [[ -z "$hotkey_cmd" ]]; then
    return
  fi

  "$hotkey_cmd" bootstrap
  local bootstrap_code=$?
  if (( bootstrap_code == 0 )); then
    mkdir -p "$conf_dir"
    : > "$sentinel"
  else
    echo "快捷键自动初始化失败（窗口整理将继续执行）。"
  fi
}

resolve_local_helper() {
  local helper_name="$1"
  local candidate=""
  local script_dir="${0:A:h}"

  candidate="$HOME/.local/bin/${helper_name}"
  if [[ -x "$candidate" ]]; then
    echo "$candidate"
    return 0
  fi

  if [[ -x "${script_dir}/${helper_name}" ]]; then
    echo "${script_dir}/${helper_name}"
    return 0
  fi

  if command -v "$helper_name" >/dev/null 2>&1; then
    command -v "$helper_name"
    return 0
  fi

  return 1
}

read_config_scope() {
  local config_file="${TILE_CONFIG_FILE:-$HOME/.config/terminal-window-tiler/config}"
  local line=""
  local key=""
  local value=""

  [[ -f "$config_file" ]] || return 0

  while IFS= read -r line || [[ -n "$line" ]]; do
    line="${line#"${line%%[![:space:]]*}"}"
    [[ -z "$line" || "$line" == \#* ]] && continue
    [[ "$line" == *"="* ]] || continue

    key="${line%%=*}"
    value="${line#*=}"
    key="${key//[[:space:]]/}"
    value="${value#"${value%%[![:space:]]*}"}"
    value="${value%"${value##*[![:space:]]}"}"

    if [[ "$key" == "TILE_SCOPE" ]]; then
      echo "$value"
      return 0
    fi
  done < "$config_file"
}

normalize_scope() {
  local value="${1:-}"
  value="${value//[[:space:]]/}"
  value="${(L)value}"
  case "$value" in
    ""|current)
      echo "current"
      ;;
    all|cross|crossterminal|cross-terminal|global)
      echo "all"
      ;;
    *)
      echo "current"
      ;;
  esac
}

on shouldSwapRecordIndices(leftRecIndex, rightRecIndex, recordList, useColumnOrder, useRightToLeft)
  set leftRec to item leftRecIndex of recordList

  set leftX to item 3 of leftRec

  set rightX to item 3 of rightRec
  set rightY to item 4 of rightRec
  set laneTolerance to 48

  if useColumnOrder then
    if my absValue(leftX - rightX) > laneTolerance then
      if useRightToLeft then
        return leftX < rightX
      else
        return leftX > rightX
      end if
    end if
    if my absValue(leftY - rightY) > laneTolerance then return leftY > rightY
    if useRightToLeft then
      return leftX < rightX
    else
      return leftX > rightX
    end if
  end if

  if my absValue(leftY - rightY) > laneTolerance then return leftY > rightY
  if my absValue(leftX - rightX) > laneTolerance then return leftX > rightX

```zsh
    if [[ "$force" == "1" ]]; then
      rm -f "$CURRENT_RUNNER_PATH"
    else
      echo "SKIP:found-unmanaged-runner:${CURRENT_RUNNER_PATH}"
      return 2
    fi
  fi

  create_runner
  chmod +x "$CURRENT_RUNNER_PATH"
  return 0
}

combo_to_key_equivalent() {
  local combo="$1"
  local raw="${combo:l}"
  raw="${raw//[[:space:]]/}"

  if [[ -z "$raw" ]]; then
    return 1
  fi

  local -a parts
  parts=("${(@s:+:)raw}")

  local has_cmd=0
  local has_shift=0
  local has_opt=0
  local has_ctrl=0
  local key=""

  local p
  for p in "${parts[@]}"; do
    case "$p" in
      cmd|command)
        (( has_cmd == 0 )) || return 1
        has_cmd=1
        ;;
      shift)
        (( has_shift == 0 )) || return 1
        has_shift=1
        ;;
      opt|option|alt)
        (( has_opt == 0 )) || return 1
        has_opt=1
        ;;
      ctrl|control)
        (( has_ctrl == 0 )) || return 1
        has_ctrl=1
        ;;
      [a-z])
        [[ -z "$key" ]] || return 1
        key="$p"
        ;;
      *)
        return 1
        ;;
    esac
  done

  [[ -n "$key" ]] || return 1
  (( has_cmd || has_shift || has_opt || has_ctrl )) || return 1

  local mods=""
  if (( has_cmd )); then
    mods="@"
    (( has_shift )) && mods="${mods}$"
    (( has_opt )) && mods="${mods}~"
    (( has_ctrl )) && mods="${mods}^"
  else
    (( has_ctrl )) && mods="${mods}^"
    (( has_opt )) && mods="${mods}~"
    (( has_shift )) && mods="${mods}$"
  fi

  local display_key="$key"

  local -a canonical_mods
  (( has_cmd )) && canonical_mods+=("cmd")
  (( has_shift )) && canonical_mods+=("shift")
  (( has_ctrl )) && canonical_mods+=("ctrl")
  (( has_opt )) && canonical_mods+=("opt")

  local canonical="${(j:+:)canonical_mods}+$key"
  local keyeq="${mods}${display_key}"
  printf '%s|%s\n' "$canonical" "$keyeq"
}

key_equivalent_to_combo() {
  local keyeq="$1"
  [[ -n "$keyeq" ]] || return 1

  local key="${keyeq[-1]}"
  local mods="${keyeq[1,-2]}"
  local has_cmd=0
  local has_shift=0
  local has_opt=0
  local has_ctrl=0

  local i char
  for (( i = 1; i <= ${#mods}; i++ )); do
    char="${mods[$i]}"
    case "$char" in
      "@") has_cmd=1 ;;
      "$") has_shift=1 ;;
      "~") has_opt=1 ;;
      "^") has_ctrl=1 ;;
    esac
  done

  local -a out
  (( has_cmd )) && out+=("cmd")
  (( has_shift )) && out+=("shift")
  (( has_ctrl )) && out+=("ctrl")
  (( has_opt )) && out+=("opt")
  out+=("${key:l}")
  printf '%s\n' "${(j:+:)out}"
}

key_equivalent_to_nsuikey() {
  local keyeq="$1"
  [[ -n "$keyeq" ]] || return 1
  printf '%s%s\n' "${keyeq[1,-2]}" "${keyeq[-1]:l}"
}

list_service_keys_with_equivalent() {
  local target="$1"

  "$PLIST_BUDDY" -c "Print NSServicesStatus" "$PBS_PLIST" 2>/dev/null | \
    awk -v target="$target" '
      /^[[:space:]]*.+ = Dict \{$/ {
        key=$0
        sub(/^[[:space:]]*/, "", key)
        sub(/ = Dict \{$/, "", key)
        next
      }
      /key_equivalent = / {
        value=$0
        sub(/^.*key_equivalent = /, "", value)
        if (value == target) print key
      }
    '
}

collect_conflicts() {
  local keyeq="$1"
  local -a collisions=()
  local service_key
  while IFS= read -r service_key; do
    [[ -z "$service_key" ]] && continue
    [[ "$service_key" == "$CURRENT_SERVICE_KEY" ]] && continue
    collisions+=("$service_key")
  done < <(list_service_keys_with_equivalent "$keyeq")

  printf '%s\n' "${collisions[@]}"
}

prompt_for_custom_combo() {
  while true; do
    printf "默认快捷键 %s 已冲突，请输入新的快捷键组合（例如 cmd+shift+t），或输入 skip 跳过： " "$CURRENT_DEFAULT_COMBO" >&2
    if ! read -r input_combo; then
      echo "SKIP"
      return 0
    fi

    if [[ "${input_combo:l}" == "skip" ]]; then
      echo "SKIP"
      return 0
    fi

    local parsed canonical keyeq
    parsed="$(combo_to_key_equivalent "$input_combo" 2>/dev/null || true)"
    if [[ -z "$parsed" ]]; then
      echo "格式无效。仅支持 cmd|shift|opt|ctrl + 单字母（a-z），例如 cmd+shift+t。" >&2
      continue
    fi

    canonical="${parsed%%|*}"
    keyeq="${parsed##*|}"

    local -a conflicts
    conflicts=()
    while IFS= read -r line; do
      [[ -n "$line" ]] && conflicts+=("$line")
    done < <(collect_conflicts "$keyeq")

    if (( ${#conflicts[@]} > 0 )); then
      echo "该快捷键已被其他 Service 占用：" >&2
      local c
      for c in "${conflicts[@]}"; do
        echo "- $c" >&2
      done
      continue
    fi

    echo "${canonical}|${keyeq}"
    return 0
  done
}

ensure_install_prerequisites() {

  install_runner "$force" >/dev/null
  local runner_code=$?
  install_workflow "$force" >/dev/null
  local workflow_code=$?

  if (( runner_code == 2 )); then
    echo "快捷键初始化跳过：检测到未托管的 runner 文件 (${CURRENT_RUNNER_PATH})。"
    return 2
  fi

  if (( workflow_code == 2 )); then
    echo "快捷键初始化跳过：检测到未托管的 Service (${CURRENT_WORKFLOW_PATH})。"
    return 2
  fi

  if (( runner_code != 0 || workflow_code != 0 )); then
    echo "快捷键初始化失败：安装 Service 或 runner 时出错。"
    return 1
  fi

  return 0
}

cmd_bootstrap() {
  if ! is_macos; then
    echo "快捷键初始化失败：仅支持 macOS。"
    return 1
  fi

  ensure_install_prerequisites 0
  local prep_code=$?
  if (( prep_code == 2 )); then
    return 0
  fi
  if (( prep_code != 0 )); then
    return 1
  fi

  refresh_services

  local current_eq current_combo
  current_eq="$(read_service_key_equivalent)"
  if [[ -n "$current_eq" ]]; then
    set_service_key_equivalent "$current_eq"
    sync_terminal_app_shortcuts "$current_eq"
    refresh_services
    if ! verify_service_binding "$current_eq"; then
      echo "快捷键初始化失败：现有 Service 绑定未写回 pbs.plist（target=${CURRENT_TARGET}, expected=${current_eq}）。"
      print_manual_repair_hint "$CURRENT_DEFAULT_COMBO"
      return 1
    fi
    current_combo="$(key_equivalent_to_combo "$current_eq" 2>/dev/null || true)"
    if [[ -n "$current_combo" ]]; then
      echo "快捷键已就绪（${current_combo}）。"
    else
      echo "快捷键已就绪（key_equivalent=${current_eq}）。"
    fi
    return 0
  fi

  local -a default_conflicts
  default_conflicts=()
  while IFS= read -r line; do
    [[ -n "$line" ]] && default_conflicts+=("$line")
  done < <(collect_conflicts "$CURRENT_DEFAULT_KEY_EQ")

  local chosen_combo="$CURRENT_DEFAULT_COMBO"
  local chosen_eq="$CURRENT_DEFAULT_KEY_EQ"

  if (( ${#default_conflicts[@]} > 0 )); then
    if [[ -t 0 && -t 1 ]]; then
      local prompt_output
      prompt_output="$(prompt_for_custom_combo)"
      if [[ "$prompt_output" == "SKIP" ]]; then
        echo "快捷键服务已安装，但未绑定按键。"
        print_manual_repair_hint "$CURRENT_DEFAULT_COMBO"
        return 2
      fi
      chosen_combo="${prompt_output%%|*}"
      chosen_eq="${prompt_output##*|}"
    else
      echo "快捷键服务已安装，但默认键冲突且当前为非交互终端，未完成绑定。"
      print_manual_repair_hint "$CURRENT_DEFAULT_COMBO"
      return 2
    fi
  fi

  set_service_key_equivalent "$chosen_eq"
  sync_terminal_app_shortcuts "$chosen_eq"
  refresh_services
  if ! verify_service_binding "$chosen_eq"; then
    echo "快捷键初始化失败：Service 绑定未写入 pbs.plist（target=${CURRENT_TARGET}, expected=${chosen_eq}）。"
    print_manual_repair_hint "$chosen_combo"
    return 1
  fi
  echo "快捷键已就绪（${chosen_combo}）。"
  return 0
}

cmd_set() {
  if ! is_macos; then
    echo "仅支持 macOS。"
    return 1
  fi

  if [[ $# -ne 1 ]]; then
    usage
    return 1
  fi

  local parsed combo keyeq

  if [[ -z "$parsed" ]]; then
    echo "快捷键格式无效：$1"
    echo "示例：cmd+shift+t / ctrl+command+t"
    return 1

  combo="${parsed%%|*}"
  keyeq="${parsed##*|}"

  ensure_install_prerequisites 0

  if (( prep_code != 0 )); then
    return 1
  fi

  local -a conflicts
  conflicts=()
  while IFS= read -r line; do
    [[ -n "$line" ]] && conflicts+=("$line")
  done < <(collect_conflicts "$keyeq")

  if (( ${#conflicts[@]} > 0 )); then
    echo "该快捷键已被其他 Service 占用："
    local c
    for c in "${conflicts[@]}"; do
      echo "- $c"

  set_service_key_equivalent "$keyeq"
  sync_terminal_app_shortcuts "$keyeq"
  refresh_services
  if ! verify_service_binding "$keyeq"; then
    echo "快捷键更新失败：Service 绑定未写入 pbs.plist（target=${CURRENT_TARGET}, expected=${keyeq}）。"

  echo "快捷键已更新：${combo}"

cmd_status() {
  if ! is_macos; then
    echo "ERROR macOS only"
    return 1
  fi

  local has_error=0

  if [[ -d "$CURRENT_WORKFLOW_PATH" ]]; then
    if workflow_is_managed; then
      echo "OK workflow: ${CURRENT_WORKFLOW_PATH}"
    else
      echo "WARN workflow exists but unmanaged: ${CURRENT_WORKFLOW_PATH}"
    fi
  else
    echo "ERROR workflow missing: ${CURRENT_WORKFLOW_PATH}"
    has_error=1
  fi

  if [[ -x "$CURRENT_RUNNER_PATH" ]]; then
    if runner_is_managed; then
      echo "OK runner: ${CURRENT_RUNNER_PATH}"
    else
      echo "WARN runner exists but unmanaged: ${CURRENT_RUNNER_PATH}"
    fi
  else
    echo "ERROR runner missing: ${CURRENT_RUNNER_PATH}"
    has_error=1
  fi

  local primary_cmd=""
  if [[ "$CURRENT_RUNNER_MODE" == "full" ]]; then
    if [[ -x "${BIN_DIR}/tile" ]]; then
      primary_cmd="${BIN_DIR}/tile"
    else
      primary_cmd="$(command -v tile 2>/dev/null || true)"
    fi
  else
    if [[ -x "${BIN_DIR}/terminal-tile-zone-picker" ]]; then
      primary_cmd="${BIN_DIR}/terminal-tile-zone-picker"
    else
      primary_cmd="$(command -v terminal-tile-zone-picker 2>/dev/null || true)"
    fi
  fi

  if [[ -n "$primary_cmd" ]]; then
    echo "OK command: ${primary_cmd}"
  else
    echo "ERROR command missing"
    has_error=1
  fi

  local current_eq current_combo
  current_eq="$(read_service_key_equivalent)"
  local services_shortcuts_present
  services_shortcuts_present="$(read_services_shortcuts_present)"
  if [[ "$services_shortcuts_present" != "true" && "$services_shortcuts_present" != "1" ]]; then
    echo "WARN services-shortcuts: disabled (ServicesShortcutsPresent=${services_shortcuts_present:-missing})"
  else
    echo "OK services-shortcuts: enabled"
  fi

  if [[ -z "$current_eq" ]]; then
    echo "WARN service-binding: missing in pbs.plist"
  else
    current_combo="$(key_equivalent_to_combo "$current_eq" 2>/dev/null || true)"
    if [[ -n "$current_combo" ]]; then
      echo "OK service-binding: ${current_combo} (${current_eq})"
    else
      echo "OK service-binding: ${current_eq}"
    fi

    local -a conflicts
    conflicts=()
    while IFS= read -r line; do
      [[ -n "$line" ]] && conflicts+=("$line")
    done < <(collect_conflicts "$current_eq")

    if (( ${#conflicts[@]} > 0 )); then
      echo "WARN conflict: also used by"
      local item
      for item in "${conflicts[@]}"; do
        echo "- $item"
      done
    else
      echo "OK conflict-check: clear in Services domain"
    fi
  fi

  local expected_nsui=""
  if [[ -n "$current_eq" ]]; then
    expected_nsui="$(key_equivalent_to_nsuikey "$current_eq" 2>/dev/null || true)"
  fi

  local -a shortcut_present=()
  local -a shortcut_match=()
  local -a shortcut_mismatch=()
  local bundle
  for bundle in "${TERMINAL_APP_BUNDLES[@]}"; do
    local bundle_name bundle_shortcut
    bundle_name="$(bundle_display_name "$bundle")"
    bundle_shortcut="$(read_bundle_shortcut "$bundle")"

    if [[ -z "$bundle_shortcut" ]]; then
      echo "WARN app-shortcut[${bundle_name}]: missing"
      continue
    fi

    shortcut_present+=("$bundle_name=${bundle_shortcut}")
    if [[ -n "$expected_nsui" && "$bundle_shortcut" == "$expected_nsui" ]]; then
      shortcut_match+=("$bundle_name")
      echo "OK app-shortcut[${bundle_name}]: ${bundle_shortcut}"
    elif [[ -n "$expected_nsui" ]]; then
      shortcut_mismatch+=("$bundle_name=${bundle_shortcut}")
      echo "WARN app-shortcut[${bundle_name}]: ${bundle_shortcut} (expected ${expected_nsui})"
    else
      echo "WARN app-shortcut[${bundle_name}]: ${bundle_shortcut} (service binding missing)"
    fi
  done

  if [[ -z "$current_eq" && ${#shortcut_present[@]} -gt 0 ]]; then
    echo "WARN state: app shortcuts exist but Service binding is missing in pbs.plist"
  elif [[ -n "$current_eq" && ${#shortcut_match[@]} == 0 ]]; then
    echo "WARN state: Service binding exists but terminal app shortcuts are missing or mismatched"
  elif [[ ${#shortcut_mismatch[@]} -gt 0 ]]; then
    echo "WARN state: some terminal app shortcuts differ from the Service binding"
  fi

  return "$has_error"
}

cmd_uninstall() {
  if ! is_macos; then

      echo "跳过未托管 runner：${CURRENT_RUNNER_PATH}"

  rm -f "$CURRENT_SENTINEL_PATH"
  echo "快捷键已卸载。若菜单未即时刷新，可注销或重启后再试。"
}

if [[ $# -lt 1 ]]; then

  exit 1
fi
```

```swift
func collectAppKitGhosttyPIDs() -> Set<pid_t> {
  var pids: Set<pid_t> = []

  for app in NSRunningApplication.runningApplications(withBundleIdentifier: "com.mitchellh.ghostty") {
    pids.insert(app.processIdentifier)
  }

  for app in NSWorkspace.shared.runningApplications {
    let localizedName = app.localizedName?.lowercased() ?? ""
    let bundleIdentifier = app.bundleIdentifier?.lowercased() ?? ""
    let bundlePath = app.bundleURL?.path.lowercased() ?? ""
    let executablePath = app.executableURL?.path.lowercased() ?? ""
    let matchesGhostty =
      localizedName.contains("ghostty") ||
      bundleIdentifier.contains("ghostty") ||
      bundlePath.contains("ghostty") ||
      executablePath.contains("ghostty")
    if matchesGhostty {
      pids.insert(app.processIdentifier)
    }
  }

  return pids
}

func collectCGWindowGhosttyPIDs() -> Set<pid_t> {
  var pids: Set<pid_t> = []
  let options: CGWindowListOption = [.optionAll]
  guard let infos = CGWindowListCopyWindowInfo(options, CGWindowID(0)) as? [[String: Any]] else {
    return pids
  }

  for info in infos {
    let owner = (info[kCGWindowOwnerName as String] as? String ?? "").lowercased()
    guard owner.contains("ghostty") else {
      continue
    }
    if let pid = info[kCGWindowOwnerPID as String] as? pid_t {
      pids.insert(pid)
    } else if let pidNumber = info[kCGWindowOwnerPID as String] as? NSNumber {
      pids.insert(pid_t(pidNumber.intValue))
    }
  }

  return pids
}

func collectProcessGhosttyPIDs() -> Set<pid_t> {
  let process = Process()
  process.executableURL = URL(fileURLWithPath: "/usr/bin/pgrep")
  process.arguments = ["-x", "ghostty"]

  let pipe = Pipe()
  process.standardOutput = pipe
  process.standardError = Pipe()

  do {
    try process.run()
  } catch {
    return []
  }

  process.waitUntilExit()
  guard process.terminationStatus == 0 else {
    return []
  }

  let data = pipe.fileHandleForReading.readDataToEndOfFile()
  let text = String(decoding: data, as: UTF8.self)
  let lines = text
    .split(whereSeparator: \.isNewline)
    .compactMap { Int32($0.trimmingCharacters(in: .whitespaces)) }

  return Set(lines)
}

func getGhosttyPIDs() -> [pid_t] {
  let allPIDs =
    collectAppKitGhosttyPIDs()
    .union(collectCGWindowGhosttyPIDs())
    .union(collectProcessGhosttyPIDs())
  return Array(allPIDs).sorted()
}

func copyAXValue(_ element: AXUIElement, attribute: String) -> AXValue? {
  var rawValue: CFTypeRef?
  let error = AXUIElementCopyAttributeValue(element, attribute as CFString, &rawValue)
  guard error == .success, let rawValue else {
    return nil
  }
  guard CFGetTypeID(rawValue) == AXValueGetTypeID() else {
    return nil
  }
  let axValue = rawValue as! AXValue
  return axValue
}

func getWindowNumber(_ element: AXUIElement) -> Int? {
  var rawValue: CFTypeRef?

  guard error == .success, let rawValue else {

func shouldSwap(_ left: WindowRecord, _ right: WindowRecord) -> Bool {
  let laneTolerance = 48
  if abs(left.y - right.y) > laneTolerance {
    return left.y > right.y
  }
  if abs(left.x - right.x) > laneTolerance {
    return left.x > right.x
  }
  return left.sourceIndex > right.sourceIndex
}

func loadWindowRecords() throws -> [WindowRecord] {
  guard AXIsProcessTrusted() else {
    throw HelperError.accessibilityDenied
  }

  let pids = getGhosttyPIDs()
  guard !pids.isEmpty else {
    throw HelperError.ghosttyNotRunning
  }

  var records: [WindowRecord] = []
  for pid in pids {
    let appElement = AXUIElementCreateApplication(pid)
    var rawWindows: CFTypeRef?
    let copyError = AXUIElementCopyAttributeValue(appElement, kAXWindowsAttribute as CFString, &rawWindows)
    guard copyError == .success else {
      continue
    }

    let windowList = rawWindows as? [AXUIElement] ?? []
    for (sourceIndex, window) in windowList.enumerated() {
      guard
        let position = getCGPoint(window, attribute: kAXPositionAttribute),
        let size = getCGSize(window, attribute: kAXSizeAttribute)
      else {
        continue
      }

      let windowNumber = getWindowNumber(window)
      let key = windowNumber.map { "\(pid):\($0)" } ?? "\(pid):idx:\(sourceIndex + 1)"

      records.append(
        WindowRecord(
          key: key,
          element: window,
          x: Int(position.x.rounded()),
          y: Int(position.y.rounded()),
          width: Int(size.width.rounded()),
          height: Int(size.height.rounded()),
          pid: pid,
          sourceIndex: sourceIndex + 1
        )
      )
    }
  }

  guard !records.isEmpty else {
    throw HelperError.windowsUnavailable
  }

  records.sort(by: shouldSwap)
  return records
}

func handleList() throws {
  let records = try loadWindowRecords()
  let output = records.map { record in
    "\(record.key)\t\(record.x)\t\(record.y)\t\(record.x + record.width)\t\(record.y + record.height)"
  }.joined(separator: "\n")
  print(output)
}

func handleMove() throws {
  let records = try loadWindowRecords()
  var operations: [(key: String, x1: Int, y1: Int, x2: Int, y2: Int)] = []

  while let line = readLine() {
    let trimmed = line.trimmingCharacters(in: .whitespacesAndNewlines)
    if trimmed.isEmpty {
      continue
    }

    let parts = trimmed.split(separator: "\t", omittingEmptySubsequences: false)
    guard parts.count == 5 else {
      throw HelperError.invalidInput("Invalid move payload: \(trimmed)")
    }

    guard
      parts[0].isEmpty == false,
      let x1 = Int(parts[1]),
      let y1 = Int(parts[2]),
      let x2 = Int(parts[3]),
      let y2 = Int(parts[4])
    else {
      throw HelperError.invalidInput("Invalid move payload: \(trimmed)")
    }

    let key = String(parts[0])

    operations.append((key, x1, y1, x2, y2))
  }

  let recordMap = Dictionary(uniqueKeysWithValues: records.map { ($0.key, $0) })

  for operation in operations {
    guard let record = recordMap[operation.key] else {
      continue
    }

    let position = CGPoint(x: operation.x1, y: operation.y1)
    let size = CGSize(width: max(operation.x2 - operation.x1, 1), height: max(operation.y2 - operation.y1, 1))

    let positionResult = setCGPoint(record.element, attribute: kAXPositionAttribute, value: position)
    let sizeResult = setCGSize(record.element, attribute: kAXSizeAttribute, value: size)
    if positionResult != .success || sizeResult != .success {
      throw HelperError.moveFailed("Failed to move Ghostty window \(operation.key): pos=\(positionResult.rawValue) size=\(sizeResult.rawValue)")
    }
  }
}

do {
  let command = CommandLine.arguments.dropFirst().first ?? "list"
  switch command {
  case "list":
    try handleList()
  case "move":
    try handleMove()
  default:
    throw HelperError.invalidInput("Usage: ghostty-window-helper [list|move]")
  }
} catch let error as HelperError {
  stderr(error.description)
  exit(1)
} catch {
  stderr(String(describing: error))
  exit(1)
}
```
